#version 330 core   

const vec3 colors[3] = vec3[](
      vec3(1.0, 0.0, 0.0),
      vec3(0.0, 1.0, 0.0),
      vec3(0.0, 0.0, 1.0)
  );

layout (location=0) in vec3 vertex_position;
layout (location=1) in uint color_index;

out vec3 fragment_color;

void main(){
      gl_Position = vec4(vertex_position, 1.0);
      fragment_color = colors[color_index];
  }
